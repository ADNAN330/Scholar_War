7.4

معالجة عناصر المصفوفة

▼

أ) جمع القيم وحساب المعدل

C++ const int NUM_SCORES = 5; double scores[NUM_SCORES] = {85, 90, 78, 92, 88}; double total = 0; for (double val : scores) total += val; double average = total / NUM_SCORES; cout << "المعدل: " << average << endl; // 86.6

ب) إيجاد الأعلى والأدنى

C++ const int SIZE = 5; int numbers[SIZE] = {42, 7, 99, 15, 33}; // إيجاد الأعلى — ابدأ بالعنصر الأول دائماً int highest = numbers[0]; for (int i = 1; i < SIZE; i++) if (numbers[i] > highest) highest = numbers[i]; cout << "الأعلى: " << highest << endl; // 99 // إيجاد الأدنى int lowest = numbers[0]; for (int i = 1; i < SIZE; i++) if (numbers[i] < lowest) lowest = numbers[i]; cout << "الأدنى: " << lowest << endl; // 7

ج) نسخ مصفوفة — لا يمكن بالإسناد المباشر!

C++ const int SIZE = 4; int oldValues[SIZE] = {10, 100, 200, 300}; int newValues[SIZE]; // ⚠️ لا يمكن: newValues = oldValues; // خطأ! // الطريقة الصحيحة — عنصراً عنصراً for (int i = 0; i < SIZE; i++) newValues[i] = oldValues[i];

د) مقارنة مصفوفتين — لا يمكن بـ ==

C++ const int SIZE = 5; int a[SIZE] = {5, 10, 15, 20, 25}; int b[SIZE] = {5, 10, 15, 20, 25}; // ⚠️ لا يمكن: if (a == b) — يقارن العناوين لا القيم! // الطريقة الصحيحة bool equal = true; int count = 0; while (equal && count < SIZE) { if (a[count] != b[count]) equal = false; count++; } if (equal) cout << "المصفوفتان متساويتان" << endl; else cout << "المصفوفتان مختلفتان" << endl;

هـ) المصفوفات الممتلئة جزئياً (Partially Filled)

C++ const int SIZE = 100; // حجم أكبر من اللازم int numbers[SIZE]; int count = 0; // عدد العناصر الفعلية int num; cout << "أدخل أرقاماً (-1 للخروج): "; cin >> num; while (num != -1 && count < SIZE) { numbers[count] = num; count++; cin >> num; } // طباعة العناصر الصالحة فقط (بـ count وليس SIZE) for (int i = 0; i < count; i++) cout << numbers[i] << endl;

المصفوفات المتوازية — Parallel Arrays

▼

المفهوم: مصفوفتان مرتبطتان بنفس الفهرس — كل عنصر في إحداهما يقابل عنصراً في الأخرى.

C++ #include <iostream> #include <iomanip> using namespace std; int main() { const int NUM_EMPLOYEES = 5; // مصفوفتان متوازيتان — نفس الفهرس = نفس الموظف int hours[NUM_EMPLOYEES] = {10, 15, 20, 40, 40}; double payRate[NUM_EMPLOYEES] = {9.75, 8.62, 10.50, 18.75, 15.65}; cout << fixed << showpoint << setprecision(2); for (int i = 0; i < NUM_EMPLOYEES; i++) { double gross = hours[i] * payRate[i]; cout << "Employee #" << (i+1) << ": $" << gross << endl; } // الخرج: // Employee #1: $97.50 // Employee #2: $129.30 // Employee #3: $210.00 // Employee #4: $750.00 // Employee #5: $626.00 return 0; }
الفكرة: hours[2] و payRate[2] كلاهما ينتميان للموظف رقم 3. الربط يتم عبر نفس الفهرس.

أ) تمرير عنصر واحد

ج) const لمنع التعديل

المصفوفات ثلاثية الأبعاد — 3D Arrays

▼

المفهوم: لا تحدّ لغة C++ عدد الأبعاد التي يمكن أن تحتويها المصفوفة. تخيّل المصفوفة ثلاثية الأبعاد كـ طوابق من الجداول: طوابق × صفوف × أعمدة.

الصيغة العامة:
type name[طوابق][صفوف][أعمدة];

مثال الكتاب: double seats[3][5][8]; — 3 طوابق، 5 صفوف، 8 أعمدة

مثال كامل مع الشرح

C++ #include <iostream> using namespace std; int main() { // seats[طوابق][صفوف][أعمدة] double seats[3][5][8]; // تخزين قيمة: الطابق 0، الصف 2، العمود 4 seats[0][2][4] = 150.0; // طباعة كل المصفوفة for (int i = 0; i < 3; i++) { // i = الطابق cout << "=== الطابق " << i << " ===" << endl; for (int j = 0; j < 5; j++) { // j = الصف for (int k = 0; k < 8; k++) { // k = العمود seats[i][j][k] = i*100 + j*10 + k; cout << seats[i][j][k] << "\t"; } cout << endl; } } return 0; }
⚠️ ملاحظة مهمة: عند تمرير مصفوفة متعددة الأبعاد لدالة، يجب تحديد جميع الأبعاد ما عدا الأول:
void show(double arr[][5][8]) — صحيح ✅
void show(double arr[][][]) — خطأ ❌

أحاديةint a[5]5
ثنائيةint a[3][4]12

مقدمة إلى STL Vector

▼

المفهوم: المتجه هو مصفوفة مرنة الحجم. يشبه المصفوفة في الناحية الوظيفية لكنه يتفوق عليها في نواح عديدة.

الفرق الجوهري:
المصفوفة: حجم ثابت يُحدد عند التعريف.
Vector: حجم مرن — يكبر عند الإضافة ويصغر عند الحذف تلقائياً.

يجب تضمين المكتبة

C++ #include <vector> // مطلوبة دائماً using namespace std; // مطلوبة للاستخدام المريح

طرق تعريف المتجه (جميع الطرق من المحاضرة)

C++ vector<int> v1; // متجه فارغ — لا عناصر vector<int> v2(10); // 10 عناصر قيمتها 0 vector<int> v3(10, 2); // 10 عناصر كلها = 2 vector<int> v4(v3); // نسخة مستقلة من v3 vector<int> v5 {10,20,30,40}; // قائمة تهيئة (C++11) vector<float> amounts; // float vector<string> names; // string vector<double> values2(values1); // نسخ من متجه آخر vector<char> letters(25,'A'); // 25 حرف كلها 'A'

تخزين وقراءة القيم بـ [] — برنامج 7-24

برنامج 7-24 #include <iostream> #include <iomanip> #include <vector> using namespace std; int main() { const int N = 5; vector<int> hours(N); vector<double> payRate(N); for (int i = 0; i < N; i++) { cout << "ساعات الموظف #" << (i+1) << ": "; cin >> hours[i]; cout << "الأجر الساعي: "; cin >> payRate[i]; } cout << fixed << showpoint << setprecision(2); for (int i = 0; i < N; i++) { double gross = hours[i] * payRate[i]; cout << "الموظف #" << (i+1) << ": $" << gross << endl; } return 0; }

حلقة for النطاقية مع المتجه — برنامج 7-25 / 7-26

برنامج 7-25 / 7-26 // قراءة فقط — x نسخة (بدون &) vector<int> numbers {10, 20, 30, 40, 50}; for (int val : numbers) cout << val << endl; // يطبع القيم بدون تعديل // تعديل القيم — x مرجع (&) vector<int> data(5); for (int &val : data) { cout << "أدخل قيمة: "; cin >> val; // يعدّل العنصر الأصلي مباشرة }
⚠️ قاعدة مهمة: بدون & تحصل على نسخة → التعديل لا يؤثر على المتجه.
مع & تحصل على مرجع → التعديل يؤثر على المتجه.

البحث الخطي — Linear Search

▼

المفهوم: خوارزمية بسيطة تفحص كل عنصر من البداية. تستخدم حلقة تسلسلية للانتقال عبر المصفوفة. لا يشترط أن تكون المصفوفة مرتبة.

الخوارزمية بلغة طبيعية (Pseudocode)

  • 1
    افترض أن القيمة غير موجودة (position = -1, found = false)
  • 2
    ابدأ من العنصر الأول (index = 0)
  • 3
    قارن list[index] مع القيمة المطلوبة
  • 4
    إذا وجدت: سجّل الموقع وأوقف البحث
  • 5
    إذا لم تجد: انتقل للعنصر التالي
  • 6
    أعد الموقع (-1 إذا لم توجد)

دالة البحث الخطي — برنامج 8-1

برنامج 8-1 int linearSearch(const int arr[], int size, int value) { int index = 0; int position = -1; // -1 = غير موجود bool found = false; while (index < size && !found) { if (arr[index] == value) { found = true; position = index; // سجّل الموقع } index++; // انتقل للتالي } return position; // أعد الموقع أو -1 } int main() { const int SIZE = 5; int tests[SIZE] = {87, 75, 98, 100, 82}; int result = linearSearch(tests, SIZE, 100); if (result == -1) cout << "لم تحصل على 100" << endl; else cout << "حصلت على 100 في الاختبار #" << (result+1) << endl; // الخرج: حصلت على 100 في الاختبار #4 }
لماذا -1؟ لأن -1 ليس فهرساً صحيحاً لأي عنصر في المصفوفة، إذن نستخدمه للإشارة لعدم الوجود.

البحث الثنائي — Binary Search

▼

المفهوم: أكثر كفاءة من الخطي. شرط وحيد: المصفوفة يجب أن تكون مرتبة! تبدأ من المنتصف وتقسّم المصفوفة لنصفين في كل خطوة.

كيف يعمل خطوة بخطوة

  • 1
    حدد first=0 و last=size-1
  • 2
    احسب middle = (first+last)/2
  • 3
    إذا array[middle] == value → وجدنا! أعد middle
  • 4
    إذا array[middle] > value → القيمة في النصف الأيسر → last = middle-1
  • 5
    إذا array[middle] < value → القيمة في النصف الأيمن → first = middle+1
  • 6
    كرر حتى تجد أو first > last

دالة البحث الثنائي — برنامج 8-2

برنامج 8-2 int binarySearch(const int array[], int size, int value) { int first = 0; int last = size - 1; int middle; int position = -1; bool found = false; while (!found && first <= last) { middle = (first + last) / 2; // نقطة المنتصف if (array[middle] == value) { // وجدنا! found = true; position = middle; } else if (array[middle] > value) // في النصف الأيسر last = middle - 1; else // في النصف الأيمن first = middle + 1; } return position; // موقع أو -1 } int main() { const int SIZE = 20; // ⚠️ يجب أن تكون مرتبة تصاعدياً int ids[SIZE] = {101,142,147,189,199,207,222, 234,289,296,310,319,388,394, 417,429,447,521,536,600}; int empID; cout << "أدخل رقم الموظف: "; cin >> empID; int result = binarySearch(ids, SIZE, empID); if (result == -1) cout << "الرقم غير موجود" << endl; else cout << "الرقم موجود في العنصر " << result << endl; }

7.11

فرز الفقاعات — Bubble Sort

▼

المفهوم: يقارن كل عنصرين متجاورين ويبدّلهما إذا كان الأول أكبر. في كل تمريرة تصل أكبر قيمة إلى مكانها الصحيح في النهاية.

تتبع خطوة بخطوة — [7, 2, 3, 8, 9, 1]

Trace المصفوفة الأولى: [7, 2, 3, 8, 9, 1] الدورة 1: مقارنة كل زوجين متجاورين [7,2] → بدّل → [2, 7, 3, 8, 9, 1] [7,3] → بدّل → [2, 3, 7, 8, 9, 1] [7,8] → لا → [2, 3, 7, 8, 9, 1] [8,9] → لا → [2, 3, 7, 8, 9, 1] [9,1] → بدّل → [2, 3, 7, 8, 1, 9] ✅ 9 في مكانه الدورة 2: [2,3] → لا [3,7] → لا [7,8] → لا [8,1] → بدّل → [2, 3, 7, 1, 8, 9] ✅ 8 في مكانه ... تستمر حتى يتم الترتيب الكامل النتيجة: [1, 2, 3, 7, 8, 9] ✅

الكود الكامل — برنامج 8-4

برنامج 8-4 void bubbleSort(int array[], int size) { int maxElement; int index; // كل دورة تضع أكبر قيمة في مكانها for (maxElement = size-1; maxElement > 0; maxElement--) { for (index = 0; index < maxElement; index++) { if (array[index] > array[index + 1]) { swap(array[index], array[index + 1]); } } } } int main() { const int SIZE = 6; int values[SIZE] = {6, 1, 5, 2, 4, 3}; cout << "قبل: "; for (auto x : values) cout << x << " "; cout << endl; bubbleSort(values, SIZE); cout << "بعد: "; for (auto x : values) cout << x << " "; // الخرج: 1 2 3 4 5 6 }

فرز الاختيار — Selection Sort

▼

المفهوم: يبحث عن أصغر قيمة في المصفوفة وينقلها للعنصر 0، ثم يبحث عن الأصغر من الباقي وينقله للعنصر 1، وهكذا.

تتبع خطوة بخطوة — [5, 7, 2, 8, 9, 1]

Trace الدورة 1: ابحث من index=0 حتى النهاية الأصغر = 1 (في index=5) → بدّل مع index=0 النتيجة: [1, 7, 2, 8, 9, 5] ✅ 1 في مكانه الدورة 2: ابحث من index=1 حتى النهاية الأصغر = 2 (في index=2) → بدّل مع index=1 النتيجة: [1, 2, 7, 8, 9, 5] ✅ 2 في مكانه الدورة 3: الأصغر = 5 → بدّل مع index=2 النتيجة: [1, 2, 5, 8, 9, 7] ✅ 5 في مكانه الدورة 4: الأصغر = 7 → النتيجة: [1, 2, 5, 7, 9, 8] الدورة 5: الأصغر = 8 → النتيجة: [1, 2, 5, 7, 8, 9] ✅

الكود الكامل — برنامج 8-5

برنامج 8-5 void swap(int &a, int &b) { int temp = a; a = b; b = temp; } void selectionSort(int array[], int size) { int minIndex, minValue; for (int start = 0; start < (size-1); start++) { minIndex = start; minValue = array[start]; // ابحث عن أصغر قيمة في الجزء المتبقي for (int index = start+1; index < size; index++) { if (array[index] < minValue) { minValue = array[index]; minIndex = index; // سجّل موقع الأصغر } } swap(array[minIndex], array[start]); // بدّله للمكان الصحيح } } int main() { const int SIZE = 6; int values[SIZE] = {6, 1, 5, 2, 4, 3}; cout << "قبل الترتيب: "; for (auto x : values) cout << x << " "; cout << endl; selectionSort(values, SIZE); cout << "بعد الترتيب: "; for (auto x : values) cout << x << " "; // الخرج: 1 2 3 4 5 6 }

الفرز والبحث على المتجهات — Sorting and Searching Vectors

▼

المفهوم: يمكن تطبيق خوارزميات الفرز والبحث التي درسناها على المتجهات وكذلك المصفوفات، بما فيها متجهات النصوص (string).

برنامج 8-7 — فرز أسماء وبحث ثنائي على vector<string>

برنامج 8-7 #include <iostream> #include <string> #include <vector> using namespace std; void selectionSort(vector<string> &v); int binarySearch(const vector<string> &v, string str); int main() { vector<string> names { "Lopez","Smith","Pike","Jones","Abernathy", "Hall","Wilson","Kimura","Alvarado","Harrison" }; selectionSort(names); // ترتيب أبجدي for (auto name : names) cout << name << endl; string searchValue; cout << "ابحث عن: "; getline(cin, searchValue); int position = binarySearch(names, searchValue); if (position != -1) cout << "موجود في الموقع: " << position << endl; else cout << "غير موجود." << endl; return 0; } // فرز الاختيار على متجه string (مقارنة أبجدية) void selectionSort(vector<string> &v) { for (int start = 0; start < (int)(v.size()-1); start++) { int minIndex = start; string minValue = v[start]; for (int i = start+1; i < (int)v.size(); i++) { if (v[i] < minValue) { // مقارنة أبجدية تلقائية minValue = v[i]; minIndex = i; } } swap(v[minIndex], v[start]); } } // البحث الثنائي على متجه string int binarySearch(const vector<string> &v, string str) { int first=0, last=(int)v.size()-1, middle, position=-1; bool found=false; while (!found && first <= last) { middle = (first+last)/2; if (v[middle] == str) { found=true; position=middle; } else if (v[middle] > str) last = middle-1; else first = middle+1; } return position; }

الأسماء المرتبة: Abernathy, Alvarado, Hall, Harrison, Jones, Kimura, Lopez, Pike, Smith, Wilson

أنواع الأخطاء — Error Types

▼
Syntax Error
خطأ في كتابة الكود. يكتشفه المُجمِّع (compiler) قبل التشغيل. البرنامج لا يُجمَّع أصلاً.
// فاصلة منقوطة مفقودة int x = 5 // ❌ خطأ تجميع
Runtime Error
الكود يُجمَّع بنجاح لكن يحدث خطأ أثناء التشغيل (crash أو استثناء).
// وصول خارج الحدود بـ at() v.at(100); // vector size=3 ❌
Logical Error
الكود يُجمَّع ويعمل لكن النتيجة خاطئة بسبب خطأ في المنطق.
// البحث عن الأكبر لكن الشرط معكوس if (a[i] < max) max = a[i]; // ❌
Undefined Behavior
الوصول لذاكرة خارج الحدود بـ []. قد يعمل أو يتعطل أو يعطي أرقام عشوائية.
// خارج حدود المصفوفة بـ [] int a[3]; cout << a[5]; // ❌ UB
الخطأمتى يظهر؟مثال
Syntax Errorوقت التجميعفاصلة منقوطة مفقودة، متغير غير معرّف
Runtime Errorوقت التشغيلat() خارج الحدود، قسمة على صفر
Logical Errorفي النتيجةشرط معكوس، حلقة خاطئة
Undefined Behaviorغير متوقع[] خارج الحدود، pop_back على فارغ
